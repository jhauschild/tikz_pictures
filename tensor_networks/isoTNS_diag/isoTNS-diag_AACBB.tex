\documentclass[tikz]{standalone}
\usepackage{tikz}
\usetikzlibrary{tensornetwork}
\usetikzlibrary{3d}
\usepackage{xifthen}

\begin{document}
\begin{tikzpicture}[z=(80:6mm),scale=0.7]
	% arrows outside center column at 2
	\foreach \x in {0,4}{
		\foreach \y in {0,2,4} {
			\ifthenelse{\x < 2}{
				\def\arrowdirection{M>}
			}{
				\def\arrowdirection{M<}
			}
			\ifthenelse{\y < 5}{\draw[tarrowleg=\arrowdirection] (\x,\y) -- +(+1,+1) ;}{}
			\ifthenelse{\y > 0}{\draw[tarrowleg=\arrowdirection] (\x,\y) -- +(+1,-1) ;}{}
		}
	} 
	\foreach \x in {1,3}{
		\foreach \y in {1,3,5} {
			\ifthenelse{\x < 2}{
				\def\arrowdirection{M>}
			}{
				\def\arrowdirection{M<}
			}
			\ifthenelse{\y < 5}{\draw[tarrowleg=\arrowdirection] (\x,\y) -- +(+1,+1) ;}{}
			\ifthenelse{\y > 0}{\draw[tarrowleg=\arrowdirection] (\x,\y) -- +(+1,-1) ;}{}
		}
	} 
	% arrows to orthogonality center
	\foreach \x in {2} {
		\foreach \y in {0,2,4}
			\draw[tarrowleg] (\x,\y) -- +(+0.5,+0.5) ;
		\foreach \y in {2,4}
			\draw[tarrowleg] (\x,\y) -- +(+0.5,-0.5) ;
	}
	\foreach \x in {3} {
		\foreach \y in {1,3}
			\draw[tarrowleg] (\x,\y) -- +(-0.5,+0.5) ;
		\foreach \y in {1,3,5}
			\draw[tarrowleg] (\x,\y) -- +(-0.5,-0.5) ;
	}
	% center column
	\foreach \y in {0.5,1.5,2.5,3.5} {
		\ifnum\fpeval{\y < 2.0}=1   % ifthenelse can only handle integers, so here we use \ifnum
			\draw[tarrowleg=M>,color=blue] (2.5,\y) -- +(0,1);
		\else
			\draw[tarrowleg=M<,color=blue] (2.5,\y) -- +(0,1);
		\fi
	}
	\foreach \x in {0,2,4} {
		\foreach \y in {0,2,4} {
			\node[smalltensor] at (\x,\y) {};
			\node[smalltensor] at ($(\x,\y) + (1,1) $){};
			\draw[tarrowleg, color=gray] (\x,\y,1) -- +(0,0,-1);
			\draw[tarrowleg, color=gray] ($(\x,\y) + (1,1,1) $) -- +(0,0,-1);
		}
	}
	\foreach \y in {0.5,1.5,...,4.5}
		\node[smalltensor=blue] at (2.5,\y){};
	\node[smalltensor=blue!60!white] at (2.5,2.5){};
	\foreach \x/\lbl in {0/A_L,1/A_L,2/A_L,
			2.5/C, 
			3/A_R, 4/A_R, 5/A_R}
		\node at (\x,-.5){$\lbl$};
\end{tikzpicture}
\end{document}

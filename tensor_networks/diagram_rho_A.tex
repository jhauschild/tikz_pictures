\begin{document}
\begin{tikzpicture}[baseline=(A.north)]
	\matrix[tmatrix]{
		\node (Ad)  [tensor,minimum width=1.2cm] {$\bar{A}$}; \&
		\node (Ld)  [tensor,shape=circle] {$\Lambda$}; \&
		\node (Bd)  [emptytensor,minimum width=1.2cm] {}; \\
		\node (A)  [tensor,minimum width=1.2cm] {$A$}; \&
		\node (L)  [tensor,shape=circle] {$\Lambda$}; \&
		\node (B)  [emptytensor,minimum width=1.2cm] {}; \\
	};
	\foreach \x in {-0.4,0.,0.4}
	{
		\draw[tedge] ($ (A.south) + (\x,0.) $) to +(0,-0.3); 
		\draw[tedge] ($ (Ad.north) + (\x,0.) $) to +(0,0.3); 
	}
	\draw[tedge] (A) to (L);
	\draw[tedge] (Ad) to (Ld);
	\draw[tedge] (L) to [in=0,out=0] (Ld);
	\coordinate (r) at ($ (B.south east) + (0,-0.3) $); 
	\coordinate (rd) at ($ (Bd.north east) + (0,+0.3) $); 
	\phantom{
		\foreach \x in {-0.4,0.,0.4}
		{
			\draw[tedge] ($ (B.south) + (\x,0.) $) to[out=270,in=180] (r) ;
			\draw[tedge] ($ (Bd.north) + (\x,0.) $) to[out=90,in=180] (rd) ;
		}
		\draw (r) to [in=0,out=0] (rd);
	}
\end{tikzpicture}

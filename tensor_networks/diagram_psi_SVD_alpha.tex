\documentclass[tikz]{standalone}
\usepackage{tikz}
\usetikzlibrary{tensornetwork}
\usepackage{braket}

\begin{document}
\begin{tikzpicture}[baseline=(A.base)]
	\matrix[tmatrix]{
		\node (A)  [tensor,minimum width=1.2cm] {$A$}; \&
		\node (L)  [tensor,shape=circle] {$\Lambda$}; \&
		\node (B)  [tensor,minimum width=1.2cm] {$B$}; \\
	};
	\foreach \j/\x/\T in {j_1/-0.4/A,j_2/0./A,j_3/+0.4/A,j_4/-0.4/B,j_5/0./B,j_6/+0.4/B}
	{
		\coordinate (s\j) at ($ (\T.south) + (\x,0.) $) ;
		\node (\j) [tindex,below] at ($ (\T.south) + (\x,-0.3) $) {$\ket{\j}$};
	}
	\node (alphaA) [left of=A] {$\ket{\alpha}_A$};
	\node (alphaB) [right of=B] {~$\ket{\alpha'}_B$};
	\begin{scope}[on background layer]
		\node (aA) [tensor=orange!60,draw,fit=(A) (j_1) (alphaA)] {};
		\node (aB) [tensor=orange!60,draw,fit=(B) (j_6) (alphaB)] {};
	\end{scope}
	\foreach \j in {j_1,j_2,j_3,j_4,j_5,j_6}
		\draw[tedge] (s\j) to (\j);
		\draw[tedge] (A) to node[tindex,above] {$\alpha$}
			(L) to node[tindex,above] {$\alpha'$}
			(B);
\end{tikzpicture}
